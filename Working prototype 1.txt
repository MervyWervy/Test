let questionPool = [];
let currentQuestion = null;
let players = {};
let answers = {};
let votes = {};
let round = 0;
let gameStarted = false;

io.on("connection", socket => {
  socket.on("startGame", () => {
  if (gameStarted) return;

  gameStarted = true;
  round = 0;
  answers = {};
  votes = {};
  questionPool = [...ALL_QUESTIONS];

  io.emit("goToGame");

  // 🚨 THIS WAS MISSING
  startRound();
});

  socket.on("joinGame", () => {
  if (!players[socket.id]) {
    players[socket.id] = { score: 0 };
  }

  if (currentQuestion) {
    socket.emit("question", currentQuestion.q);
  }
});


  socket.on("submitAnswer", ans => {
    answers[socket.id] = ans;
    socket.emit("submitted");

    if (Object.keys(answers).length === Object.keys(players).length) {
      let choices = Object.values(answers);
      choices.push(currentQuestion.a);
      choices.sort(() => Math.random() - 0.5);
      io.emit("showChoices", choices);
    }
  });

  socket.on("vote", choice => {
    votes[socket.id] = choice;
    socket.emit("voted");

    if (Object.keys(votes).length === Object.keys(players).length) {
      scoreRound();
    }
  });

  socket.on("disconnect", () => {
    delete players[socket.id];
    delete answers[socket.id];
    delete votes[socket.id];
  });

});

function startRound() {
  answers = {};
  votes = {};

  const index = Math.floor(Math.random() * questionPool.length);
  currentQuestion = questionPool.splice(index, 1)[0];

  io.emit("question", currentQuestion.q);
  console.log("Starting round with question:", currentQuestion.q);
}


function scoreRound() {
  let reveal = [];

  for (let voter in votes) {
    const picked = votes[voter];

    if (picked === currentQuestion.a) {
      players[voter].score += 100;
      reveal.push("✔ Someone got it RIGHT!");
    } else {
      for (let liar in answers) {
        if (answers[liar] === picked) {
          players[liar].score += 50;
          reveal.push("😈 Someone was fooled!");
        }
      }
    }
  }

  round++;

  if (round >= MAX_ROUNDS) {
    io.emit("finalScores", players);
  } else {
    io.emit("reveal", {
      correct: currentQuestion.a,
      reveal,
      scores: players
    });

    setTimeout(() => startRound(), 4000);
  }
}
